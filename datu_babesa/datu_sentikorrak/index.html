    3.1 Gestión y protección de datos sensibles¶

    Cifrado de datos¶

    ¿Por qué es importante usar cifrado?¶

    • Si los datos son robados, no podrán ser leídos
    • Garantiza la seguridad durante la transferencia de datos
    • Cumplimiento legal (GDPR, HIPAA, etc.)

    Tipos de cifrado¶

    1. Cifrado simétrico (AES)¶

    Utiliza la misma clave para cifrar y descifrar.

    Ejemplo en Python (PyCryptodome):

    from Crypto.Cipher import AES
from Crypto.Random import get_random_bytes
from Crypto.Util.Padding import pad, unpad
import base64

def cifrar(mensaje, clave=None):
    if clave is None:
        clave = get_random_bytes(32)  # Clave de 256 bits

    # Crear vector de inicialización (IV) aleatorio
    iv = get_random_bytes(16)

    # Crear el cifrador
    cipher = AES.new(clave, AES.MODE_CBC, iv)

    # Cifrar el mensaje (se debe rellenar para bloques de 16 bytes)
    mensaje_rellenado = pad(mensaje.encode('utf-8'), AES.block_size)
    texto_cifrado = cipher.encrypt(mensaje_rellenado)

    # Devolver en formato base64 (IV + texto cifrado)
    return base64.b64encode(iv + texto_cifrado).decode('utf-8')

descifrar(mensaje_cifrado, clave):
    try:
        # Decodificar base64
        datos = base64.b64decode(mensaje_cifrado)

        # Extraer IV y texto cifrado
        iv = datos[:16]
        texto_cifrado = datos[16:]

        # Crear el cifrador
        cipher = AES.new(clave, AES.MODE_CBC, iv)

        # Descifrar y eliminar el relleno
        mensaje_rellenado = cipher.decrypt(texto_cifrado)
        mensaje = unpad(mensaje_rellenado, AES.block_size).decode('utf-8')

        return mensaje
    except Exception as e:
        print(f"Error al descifrar: {e}")
        return None

# Cómo usarlo
clave = get_random_bytes(32)
mensaje = "Datos sensibles"
cifrado = cifrar(mensaje, clave)
descifrado = descifrar(cifrado, clave)

print(f"Original: {mensaje}")
print(f"Cifrado: {cifrado}")
print(f"Descifrado: {descifrado}")

    2. Cifrado asimétrico (RSA)¶

    Utiliza claves públicas y privadas diferentes.

    Ejemplo en Python (cryptography):

    from cryptography.hazmat.primitives import serialization
from cryptography.hazmat.primitives.asymmetric import rsa, padding
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.backends import default_backend
import os

def generar_claves(ruta_guardado=None):
    # Generar claves privada y pública
    privada = rsa.generate_private_key(
        public_exponent=65537,
        key_size=4096,
        backend=default_backend()
    )
    publica = privada.public_key()

    # Guardar claves en archivos (opcional)
    if ruta_guardado:
        # Guardar clave privada
        with open(f"{ruta_guardado}/clave_privada.pem", "wb") as f:
            f.write(
                privada.private_bytes(
                    encoding=serialization.Encoding.PEM,
                    format=serialization.PrivateFormat.PKCS8,
                    encryption_algorithm=serialization.NoEncryption()
                )
            )

        # Guardar clave pública
        with open(f"{ruta_guardado}/clave_publica.pem", "wb") as f:
            f.write(
                publica.public_bytes(
                    encoding=serialization.Encoding.PEM,
                    format=serialization.PublicFormat.SubjectPublicKeyInfo
                )
            )

    return privada, publica

def cifrar_rsa(mensaje, clave_publica):
    # Cifrar con la clave pública
    texto_cifrado = clave_publica.encrypt(
        mensaje.encode('utf-8'),
        padding.OAEP(
            mgf=padding.MGF1(algorithm=hashes.SHA256()),
            algorithm=hashes.SHA256(),
            label=None
        )
    )
    return texto_cifrado

def descifrar_rsa(texto_cifrado, clave_privada):
    # Descifrar con la clave privada
    mensaje = clave_privada.decrypt(
        texto_cifrado,
        padding.OAEP(
            mgf=padding.MGF1(algorithm=hashes.SHA256()),
            algorithm=hashes.SHA256(),
            label=None
        )
    )
    return mensaje.decode('utf-8')

# Cómo usarlo
privada, publica = generar_claves()
mensaje = "Datos sensibles"

# Cifrar
texto_cifrado = cifrar_rsa(mensaje, publica)
print(f"Cifrado: {texto_cifrado.hex()}")
testu_zifratua = zifratu_rsa(mezua, publikoa)
print(f"Zifratua: {testu_zifratua.hex()}")

# Deszifratu
deszifratua = deszifratu_rsa(testu_zifratua, pribatua)
print(f"Deszifratua: {deszifratua}")

    Transferencia de datos (TLS/HTTPS)¶

    Cómo configurar HTTPS en Node.js (Express)¶

    const https = require('https');
const fs = require('fs');
const express = require('express');

const app = express();

// Configurar medidas de seguridad
app.use(helmet());

// HSTS (HTTP Strict Transport Security)
app.use(helmet.hsts({
  maxAge: 31536000, // 1 año
  includeSubDomains: true,
  preload: true
}));

// Otras cabeceras de seguridad
app.use(helmet.frameguard({ action: 'deny' }));
app.use(helmet.xssFilter());
app.use(helmet.noSniff());

// Leer claves SSL/TLS
const options = {
  key: fs.readFileSync('ruta/a/clave-privada.key'),
  cert: fs.readFileSync('ruta/a/certificado.crt'),
  // Opciones adicionales
  minVersion: 'TLSv1.2',
  ciphers: [
    'ECDHE-ECDSA-AES256-GCM-SHA384',
    'ECDHE-RSA-AES256-GCM-SHA384',
    // ...
  ].join(':')
};

// Iniciar servidor HTTPS
https.createServer(options, app).listen(443, () => {
  console.log('Servidor HTTPS escuchando en el puerto 443...');
});

    Consejos para el almacenamiento de datos¶

    1. No almacenes datos sensibles si no es necesario¶

    • No guardes contraseñas, sino versiones con hash
    • No guardes números de tarjetas de crédito si no es estrictamente necesario

    2. Minimización de datos¶

    • Al recopilar: Recopila solo los datos mínimos necesarios
    • Al almacenar: Guarda solo los datos que necesitas
    • Al expirar: Elimina los datos cuando ya no sean necesarios

    3. Enmascaramiento de datos para visualización¶

    function enmascararNumeroCuenta(numeroCuenta) {
  if (!numeroCuenta) return '';
  const longitud = numeroCuenta.length;
  if (longitud <= 4) return numeroCuenta;
  return '*'.repeat(longitud - 4) + numeroCuenta.slice(-4);
}

// Ejemplo
console.log(enmascararNumeroCuenta('1234567890123456')); // "************3456"

    Gestión segura de tokens¶

    Almacenamiento seguro de tokens JWT¶

    Ejemplo (usando localStorage en React - NO recomendado):

    // Forma MUY SIMPLE de guardar el token (no es segura)
const guardarToken = (token) => {
  try {
    localStorage.setItem('token', token);
  } catch (error) {
    console.error('No se pudo guardar el token:', error);
  }
};

    Mejor opción (usando cookies HTTP-only en el servidor):

    // Respuesta del servidor (Node.js + Express)
res.cookie('token', token, {
  httpOnly: true,
  secure: true, // Solo a través de HTTPS
  sameSite: 'strict',
  maxAge: 24 * 60 * 60 * 1000 // 1 día
});

    Expiración de tokens¶

    // Al crear un token (Node.js)
function crearToken(usuario) {
  return jwt.sign(
    {
      sub: usuario.id,
      iat: Math.floor(Date.now() / 1000), // Momento de creación
      exp: Math.floor(Date.now() / 1000) + (60 * 60), // Expira en 1 hora
      // Otros campos...
    },
    process.env.JWT_SECRET,
    { algorithm: 'HS256' }
  );
}

// Verificar token en el cliente (React)
useEffect(() => {
  const verificarToken = async () => {
    try {
      const respuesta = await fetch('/api/verificar-token', {
        credentials: 'include' // Para enviar la cookie
      });

      if (!respuesta.ok) {
        // Token inválido, cerrar sesión
        await cerrarSesion();
      }
    } catch (error) {
      console.error('Error al verificar el token:', error);
    }
  };

  // Verificar token periódicamente
  const intervalo = setInterval(verificarToken, 5 * 60 * 1000); // Cada 5 minutos

  return () => clearInterval(intervalo);
}, []);
