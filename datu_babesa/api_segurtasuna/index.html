    3.2 APIs Seguras y Autenticación¶

    OAuth 2.0 y OpenID Connect¶

    Flujo Completo de OAuth 2.0 (Authorization Code Flow + PKCE)¶

    sequenceDiagram
    participant Usuario
    participant Aplicación_Cliente
    participant Servidor_Autorización
    participant Servidor_Recursos

    Usuario->>Aplicación_Cliente: Intenta acceder
    Aplicación_Cliente->>Usuario: Redirige a /login
    Usuario->>Servidor_Autorización: Introduce credenciales
    Servidor_Autorización->>Usuario: Autenticado, redirige a callback
    Aplicación_Cliente->>Servidor_Autorización: Intercambia código por token
    Servidor_Autorización-->>Aplicación_Cliente: Devuelve access_token y refresh_token
    Aplicación_Cliente->>Servidor_Recursos: Petición con access_token
    Servidor_Recursos-->>Aplicación_Cliente: Devuelve los datos

    Implementación de PKCE (Proof Key for Code Exchange) (React + Node.js)¶

    Lado del Cliente (React):

    // Funciones auxiliares para generar código PKCE
const generateRandomString = (length) => {
  const array = new Uint32Array(length);
  window.crypto.getRandomValues(array);
  return Array.from(array, dec => ('0' + dec.toString(16)).substr(-2)).join('');
};

const sha256 = async (plain) => {
  const encoder = new TextEncoder();
  const data = encoder.encode(plain);
  const hash = await window.crypto.subtle.digest('SHA-256', data);
  return Array.from(new Uint8Array(hash))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');
};

const base64URLEncode = (str) => {
  return btoa(String.fromCharCode.apply(null, new Uint8Array(str)))
    .replace(/\+/g, '-')
    .replace(/\//g, '_')
    .replace(/=+$/, '');
};

// Función de inicio de sesión
const handleLogin = async () => {
  // 1. Crear code_verifier y code_challenge
  const codeVerifier = generateRandomString(64);
  const hashed = await sha256(codeVerifier);
  const codeChallenge = base64URLEncode(hashed);

  // 2. Guardar code_verifier en la sesión
  sessionStorage.setItem('code_verifier', codeVerifier);

  // 3. Redirigir al usuario al endpoint de autorización
  const authUrl = new URL('https://auth.tu-servidor.com/authorize');
  const params = {
    response_type: 'code',
    client_id: 'tu_client_id',
    redirect_uri: window.location.origin + '/callback',
    code_challenge: codeChallenge,
    code_challenge_method: 'S256',
    scope: 'openid profile email',
    state: generateRandomString(16)
  };

  Object.keys(params).forEach(key => 
    authUrl.searchParams.append(key, params[key])
  );

  window.location.href = authUrl.toString();
};

    Lado del Servidor (Node.js + Express):

    const express = require('express');
const { createHmac } = require('crypto');
const jwt = require('jsonwebtoken');

const app = express();
app.use(express.json());

// Almacenamiento en memoria (en producción usar una base de datos)
const codeVerifiers = new Map();
const refreshTokens = new Map();

// 1. Devolver código de autorización
app.get('/authorize', (req, res) => {
  const { client_id, redirect_uri, code_challenge, state } = req.query;

  // Verificar cliente y URI de redirección
  if (client_id !== 'tu_client_id') {
    return res.status(400).send('ID de cliente no válido');
  }

  // Guardar code_challenge (normalmente en una sesión)
  const authCode = generateRandomString(32);
  codeVerifiers.set(authCode, {
    codeChallenge: code_challenge,
    clientId: client_id,
    redirectUri: redirect_uri,
    createdAt: Date.now()
  });

  // Mostrar página de consentimiento al usuario y luego...
  // Después de que el usuario dé su consentimiento, redirigir al callback
  const redirectUrl = new URL(redirect_uri);
  redirectUrl.searchParams.append('code', authCode);
  redirectUrl.searchParams.append('state', state);

  res.redirect(redirectUrl.toString());
});

// 2. Intercambiar código por token
app.post('/token', async (req, res) => {
  const { client_id, code, code_verifier, redirect_uri, grant_type } = req.body;

  // Validar parámetros
  if (grant_type !== 'authorization_code') {
    return res.status(400).json({ error: 'Tipo de concesión no válido' });
  }

  // Validar código
  const codeData = codeVerifiers.get(code);
  if (!codeData) {
    return res.status(400).json({ error: 'Código no válido' });
  }

  // Validar code_verifier
  const hashed = await sha256(code_verifier);
  const codeChallenge = base64URLEncode(hashed);

  if (codeChallenge !== codeData.codeChallenge) {
    return res.status(400).json({ error: 'Verificador de código incorrecto' });
  }

  // Crear tokens
  const accessToken = jwt.sign(
    { sub: 'user123', client_id },
    process.env.JWT_SECRET,
    { expiresIn: '1h' }
  );

  const refreshToken = generateRandomString(64);
  refreshTokens.set(refreshToken, { 
    sub: 'user123',
    client_id,
    expiresAt: Date.now() + 30 * 24 * 60 * 60 * 1000 // 30 días
  });

  // Eliminar código para evitar reutilización
  codeVerifiers.delete(code);

  // Devolver tokens
  res.json({
    access_token: accessToken,
    token_type: 'Bearer',
    expires_in: 3600,
    refresh_token: refreshToken,
    id_token: '...' // Para OpenID Connect
  });
});

// 3. Verificar token (Servidor de Recursos)
function authenticateToken(req, res, next) {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  if (!token) return res.sendStatus(401);

  jwt.verify(token, process.env.JWT_SECRET, (err, user) => {
    if (err) return res.sendStatus(403);
    req.user = user;
    next();
  });
}

app.get('/api/datos', authenticateToken, (req, res) => {
  // Devolver datos del usuario
  res.json({ 
    message: 'Datos obtenidos correctamente',
    user: req.user 
  });
});

// Funciones auxiliares
function generarCadenaAleatoria(length) {
  return require('crypto').randomBytes(length).toString('hex');
}

async function sha256(plain) {
  const encoder = new TextEncoder();
  const data = encoder.encode(plain);
  const hash = await crypto.subtle.digest('SHA-256', data);
  return Buffer.from(hash);
}

function codificarBase64URL(buffer) {
  return buffer.toString('base64')
    .replace(/\+/g, '-')
    .replace(/\//g, '_')
    .replace(/=+$/, '');
}

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Zerbitzaria ${PORT} portuan entzuten...`);
});

    Preferencias de Seguridad para APIs (Security Headers)¶

    Uso de Helmet.js (Node.js)¶

    const express = require('express');
const helmet = require('helmet');

const app = express();

// Medidas de seguridad generales
app.use(helmet());

// Política de Seguridad de Contenido (CSP)
app.use(
  helmet.contentSecurityPolicy({
    directives: {
      defaultSrc: ["'self'"],
      scriptSrc: [
        "'self'",
        "'unsafe-inline'", // Eliminar esto lo antes posible
        "https://apis.google.com"
      ],
      styleSrc: ["'self'", "'unsafe-inline'"],
      imgSrc: ["'self'", "data:", "https://*.example.com"],
      connectSrc: ["'self'", "https://api.tu-servidor.com"],
      fontSrc: ["'self'", "https://fonts.gstatic.com"],
      objectSrc: ["'none'"],
      upgradeInsecureRequests: []
    }
  })
);

// Protección XSS
app.use(helmet.xssFilter());

// Prevenir MIME-type sniffing
app.use(helmet.noSniff());

// Prevenir clickjacking
app.use(helmet.frameguard({ action: 'deny' }));

// HSTS (Forzar HTTPS)
const oneYearInSeconds = 31536000;
app.use(helmet.hsts({
  maxAge: oneYearInSeconds,
  includeSubDomains: true,
  preload: true
}));

// Ocultar X-Powered-By
app.use(helmet.hidePoweredBy());

// Control de precarga DNS
app.use(helmet.dnsPrefetchControl({ allow: false }));

// Opciones de descarga (IE8+)
app.use(helmet.ieNoOpen());

// Opciones de tipo de contenido
app.use(helmet.noSniff());

// Políticas cruzadas de dominios permitidos
app.use(helmet.permittedCrossDomainPolicies());

// Opciones de marco (previene clickjacking)
app.use(helmet.frameguard({ action: 'deny' }));

// Filtro XSS (obsoleto, pero para compatibilidad)
app.use((req, res, next) => {
  res.setHeader('X-XSS-Protection', '1; mode=block');
  next();
});

    Mejores Prácticas de Seguridad para APIs (¡Muy importante!)¶

    1. Validación de Entrada¶

    // Usando express-validator
const { body, validationResult } = require('express-validator');

app.post('/api/usuarios', [
  // Nombre: 2-50 caracteres, solo letras y espacios
  body('nombre')
    .trim()
    .isLength({ min: 2, max: 50 })
    .matches(/^[\p{L}\s]+$/u)
    .withMessage('El nombre debe tener entre 2 y 50 caracteres y solo puede contener letras y espacios'),

  // Email: debe ser un correo electrónico válido
  body('email')
    .isEmail()
    .normalizeEmail()
    .withMessage('Por favor, introduce una dirección de correo electrónico válida'),

  // Contraseña: 8+ caracteres, al menos una mayúscula, una minúscula, un número y un carácter especial
  body('contrasena')
    .isStrongPassword({
      minLength: 8,
      minLowercase: 1,
      minUppercase: 1,
      minNumbers: 1,
      minSymbols: 1
    })
    .withMessage('La contraseña debe tener al menos 8 caracteres, incluyendo una letra mayúscula, una minúscula, un número y un carácter especial')
], async (req, res) => {
  // Verificar resultados de validación
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }

  // Continuar con el procesamiento
  try {
    // Crear usuario...
    res.status(201).json({ message: 'Usuario creado correctamente' });
  } catch (error) {
    console.error('Error al crear el usuario:', error);
    res.status(500).json({ error: 'Error en el servidor' });
  }
});

// 2. Límite de Tasa (Rate Limiting)

const rateLimit = require('express-rate-limit');

// Límite de tasa básico
const apiLimiter = rateLimit({
  windowMs: 15 * 60 * 1000, // Ventana de 15 minutos
  max: 100, // Número máximo de solicitudes por ventana
  message: { error: 'Demasiadas solicitudes, por favor inténtalo de nuevo más tarde' },
  standardHeaders: true, // Incluir información de límite en las cabeceras `RateLimit-*`
  legacyHeaders: false, // Deshabilitar las cabeceras `X-RateLimit-*`
});

// Aplicar límite a la API
app.use('/api/', apiLimiter);

// Límite de tasa basado en el usuario
const userLimiter = rateLimit({
  windowMs: 60 * 60 * 1000, // 1 hora
  max: async (req, res) => {
    // Obtener ID de usuario (por ejemplo, del token)
    const userId = req.user?.id || req.ip;

    // Obtener plan del usuario (por ejemplo, de una base de datos)
    const userPlan = await getUserPlan(userId);

    // Devolver límite según el plan
    switch (userPlan) {
      case 'free': return 100;
      case 'pro': return 1000;
      case 'enterprise': return 10000;
      default: return 100;
    }
  },
  keyGenerator: (req) => req.user?.id || req.ip,
  message: { error: 'Tu plan de usuario tiene un límite de solicitudes por hora' }
});

app.use('/api/usuario/datos', userLimiter);

// 3. Manejo Seguro de Errores

// Manejador de errores global
app.use((err, req, res, next) => {
  console.error(err.stack);

  // Mensaje de error genérico
  const response = {
    error: {
      message: 'Error en el servidor',
      code: 'internal_server_error',
      requestId: req.id // Usando una cabecera request-id
    }
  };

  // Información adicional en desarrollo
  if (process.env.NODE_ENV === 'development') {
    response.error.details = {
      message: err.message,
      stack: err.stack,
      ...err
    };
  }

  // Código de estado HTTP
  const statusCode = err.statusCode || 500;

  // Enviar respuesta
  res.status(statusCode).json(response);
});

// Manejo de 404
app.use((req, res) => {
  res.status(404).json({
    error: {
      message: 'No se encontró el recurso solicitado',
      code: 'not_found',
      path: req.path
    }
  });
});
